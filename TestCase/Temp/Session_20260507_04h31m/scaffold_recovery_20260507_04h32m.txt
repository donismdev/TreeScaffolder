void UMercenaryInfoReturnWidget::BuildWidgetTree()
{
	if (WidgetTree == nullptr)
	{
		return;
	}

	RootVerticalBox = WidgetTree->ConstructWidget<UVerticalBox>(UVerticalBox::StaticClass(), TEXT("ReturnRootVerticalBox"));
	if (RootVerticalBox == nullptr)
	{
		return;
	}

	WidgetTree->RootWidget = RootVerticalBox;

	const FLinearColor HeaderColor(0.96f, 0.84f, 0.42f, 1.0f);
	const FLinearColor NormalColor(0.88f, 0.88f, 0.92f, 1.0f);

	Row_ReturnPressureHeader = AddRow(HeaderColor);
	Row_ReturnOpportunityCount = AddRow(NormalColor);
	Row_ReturnRefusalCount = AddRow(NormalColor);
	Row_ReturnPressureLevel = AddRow(NormalColor);
	Row_MissingPenaltyLevel = AddRow(NormalColor);
	Row_InjuryRiskLevel = AddRow(NormalColor);
	Row_CurseRiskLevel = AddRow(NormalColor);
	Row_LootLossRiskLevel = AddRow(NormalColor);

	Row_ProgressDirectionHeader = AddRow(HeaderColor);
	Row_ProgressDirectionCounts = AddRow(NormalColor);
	Row_ProgressContextCounts = AddRow(NormalColor);
	Row_LastProgressDirection = AddRow(NormalColor);

	Row_BreakDownHeader = AddRow(HeaderColor);
	Row_BreakDownState = AddRow(NormalColor);
	Row_BreakDownReached = AddRow(NormalColor);

	Row_RouteDataHeader = AddRow(HeaderColor);
	Row_RouteDataDummy = AddRow(NormalColor);
	Row_RouteDataMission = AddRow(NormalColor);
	Row_RouteDataBurden = AddRow(NormalColor);
	Row_RouteDataRelic = AddRow(NormalColor);

	DebugReturnButton = WidgetTree->ConstructWidget<UButton>(UButton::StaticClass(), TEXT("DebugReturnButton"));
	DebugReturnButtonText = WidgetTree->ConstructWidget<UTextBlock>(UTextBlock::StaticClass(), TEXT("DebugReturnButtonText"));

	if (DebugReturnButton != nullptr && DebugReturnButtonText != nullptr)
	{
		DebugReturnButtonText->SetText(PlayCoreDisplayText::UI(TEXT("테스트 귀환 팝업")));
		DebugReturnButtonText->SetColorAndOpacity(FSlateColor(FLinearColor::White));

		FSlateFontInfo FontInfo = DebugReturnButtonText->GetFont();
		FontInfo.Size = 13;
		DebugReturnButtonText->SetFont(FontInfo);

		DebugReturnButton->SetContent(DebugReturnButtonText);
		DebugReturnButton->OnClicked.AddDynamic(this, &UMercenaryInfoReturnWidget::HandleDebugReturnClicked);

		if (UVerticalBoxSlot* DebugButtonSlot = RootVerticalBox->AddChildToVerticalBox(DebugReturnButton))
		{
			DebugButtonSlot->SetPadding(FMargin(0.0f, 4.0f, 0.0f, 2.0f));
			DebugButtonSlot->SetSize(FSlateChildSize(ESlateSizeRule::Automatic));
		}
	}
}

void UMercenaryInfoReturnWidget::RefreshWidget()
{
	if (RootVerticalBox == nullptr)
	{
		return;
	}

	const FMercenaryReturnPressureState EmptyPressure;
	const FMercenaryReturnPressureState& Pressure = SnapshotPtr != nullptr
		? SnapshotPtr->ReturnPressureState
		: EmptyPressure;

	SetRow(Row_ReturnPressureHeader.Get(), TEXT("── 귀환 압력 ──"));
	SetRow(Row_ReturnOpportunityCount.Get(), FString::Printf(TEXT("귀환 기회: %d"), Pressure.ReturnOpportunityCount));
	SetRow(Row_ReturnRefusalCount.Get(), FString::Printf(TEXT("귀환 거부: %d"), Pressure.ReturnRefusalCount));
	SetRow(Row_ReturnPressureLevel.Get(), FString::Printf(TEXT("압력: %s"), *BuildReturnPressureLevelLabel(Pressure.PressureLevel)));
	SetRow(Row_MissingPenaltyLevel.Get(), FString::Printf(TEXT("실종 약화 위험: %d"), Pressure.MissingPenaltyLevel));
	SetRow(Row_InjuryRiskLevel.Get(), FString::Printf(TEXT("영구 부상 위험: %d"), Pressure.InjuryRiskLevel));
	SetRow(Row_CurseRiskLevel.Get(), FString::Printf(TEXT("저주 위험: %d"), Pressure.CurseRiskLevel));
	SetRow(Row_LootLossRiskLevel.Get(), FString::Printf(TEXT("전리품 손실 위험: %d"), Pressure.LootLossRiskLevel));

	SetRow(Row_ProgressDirectionHeader.Get(), TEXT("── 진행 방향 ──"));

	const FMercenaryProgressDirectionState EmptyProgressDirection;
	const FMercenaryProgressDirectionState& ProgressDirection = SnapshotPtr != nullptr
		? SnapshotPtr->ProgressDirectionState
		: EmptyProgressDirection;

	SetRow(
		Row_ProgressDirectionCounts.Get(),
		FString::Printf(
			TEXT("진행 %d / 퇴각 %d / 던전 진행 %d / 던전 탈출 %d"),
			ProgressDirection.ForwardCount,
			ProgressDirection.RetreatCount,
			ProgressDirection.DungeonForwardCount,
			ProgressDirection.DungeonExitCount
		)
	);

	SetRow(
		Row_ProgressContextCounts.Get(),
		FString::Printf(
			TEXT("일반 %d / 던전 %d / 미션 %d / 이벤트 %d / 귀환루트 %d"),
			ProgressDirection.NormalContextCount,
			ProgressDirection.DungeonContextCount,
			ProgressDirection.MissionContextCount,
			ProgressDirection.EventContextCount,
			ProgressDirection.ReturnRouteContextCount
		)
	);

	SetRow(
		Row_LastProgressDirection.Get(),
		FString::Printf(
			TEXT("마지막: %s / %s"),
			*BuildProgressDirectionLabel(ProgressDirection.LastDirection),
			*BuildProgressContextLabel(ProgressDirection.LastContext)
		)
	);

	SetRow(Row_BreakDownHeader.Get(), TEXT("── Breakdown ──"));

	const FMercenaryAchievementState EmptyAchievement;
	const FMercenaryAchievementState& Achievement = SnapshotPtr != nullptr
		? SnapshotPtr->AchievementState
		: EmptyAchievement;

	SetRow(
		Row_BreakDownState.Get(),
		FString::Printf(
			TEXT("현재 %d / 최대 %d / 발생 %d회"),
			Achievement.BreakDownLevel,
			Achievement.MaxBreakDownLevel,
			Achievement.BreakDownTriggerCount
		)
	);

	SetRow(
		Row_BreakDownReached.Get(),
		FString::Printf(
			TEXT("10단 도달: %s / 20단 도달: %s"),
			Achievement.bReachedBreakDown10 == true ? TEXT("true") : TEXT("false"),
			Achievement.bReachedBreakDown20 == true ? TEXT("true") : TEXT("false")
		)
	);

	SetRow(Row_RouteDataHeader.Get(), TEXT("── RouteData ──"));

	if (SnapshotPtr == nullptr)
	{
		SetRow(Row_RouteDataDummy.Get(), TEXT("RouteData 없음"));
		SetRow(Row_RouteDataMission.Get(), TEXT(""));
		SetRow(Row_RouteDataBurden.Get(), TEXT(""));
		SetRow(Row_RouteDataRelic.Get(), TEXT(""));

		if (DebugReturnButton != nullptr)
		{
			DebugReturnButton->SetIsEnabled(false);
		}

		return;
	}

	const FMercenaryMissionProgress& MajorMission = SnapshotPtr->MissionState.MajorMission;

	SetRow(
		Row_RouteDataDummy.Get(),
		FString::Printf(
			TEXT("DummyRouteData: %s / BuildSeed: %s"),
			SnapshotPtr->bDummyRouteData == true ? TEXT("true") : TEXT("false"),
			*SnapshotPtr->BuildTraitSeedId.ToString()
		)
	);

	SetRow(
		Row_RouteDataMission.Get(),
		FString::Printf(
			TEXT("Mission: %s / Target: %s / Faction: %s / Route: %s"),
			*MajorMission.MissionId.ToString(),
			*MajorMission.TargetId.ToString(),
			*MajorMission.FactionId.ToString(),
			*UEnum::GetValueAsString(MajorMission.RouteResultState)
		)
	);

	SetRow(
		Row_RouteDataBurden.Get(),
		FString::Printf(
			TEXT("Burden: %s / Kind: %s"),
			*SnapshotPtr->PersonalBurdenState.BurdenId.ToString(),
			*UEnum::GetValueAsString(SnapshotPtr->PersonalBurdenState.Kind)
		)
	);

	SetRow(
		Row_RouteDataRelic.Get(),
		FString::Printf(
			TEXT("Relic: %s"),
			*SnapshotPtr->AchievementState.RelicId.ToString()
		)
	);

	if (DebugReturnButton != nullptr)
	{
		DebugReturnButton->SetIsEnabled(true);
	}
}

void UMercenaryInfoMissionWidget::BuildWidgetTree()
{
	if (WidgetTree == nullptr)
	{
		return;
	}

	Row_MajorMissionRows.Reset();
	Row_MinorMissionRows.Reset();

	RootVerticalBox = WidgetTree->ConstructWidget<UVerticalBox>(UVerticalBox::StaticClass(), TEXT("MissionRootVerticalBox"));
	if (RootVerticalBox == nullptr)
	{
		return;
	}

	WidgetTree->RootWidget = RootVerticalBox;

	const FLinearColor HeaderColor(0.96f, 0.84f, 0.42f, 1.0f);
	const FLinearColor NormalColor(0.88f, 0.88f, 0.92f, 1.0f);

	Row_MajorHeader = AddRow(HeaderColor);
	for (int32 Index = 0; Index < 5; ++Index)
	{
		Row_MajorMissionRows.Add(AddRow(NormalColor));
	}

	Row_MinorHeader = AddRow(HeaderColor);
	Row_MinorSummary = AddRow(NormalColor);

	for (int32 Index = 0; Index < MinorMissionRowCount; ++Index)
	{
		Row_MinorMissionRows.Add(AddRow(NormalColor));
	}
}

void UMercenaryInfoMissionWidget::RefreshWidget()
{
	if (RootVerticalBox == nullptr)
	{
		return;
	}

	SetRow(Row_MajorHeader.Get(), TEXT("── 대미션 ──"));

	if (SnapshotPtr == nullptr)
	{
		if (Row_MajorMissionRows.IsValidIndex(0) == true)
		{
			SetRow(Row_MajorMissionRows[0].Get(), TEXT("대미션 데이터 없음"));
		}

		SetRow(Row_MinorHeader.Get(), TEXT("── 자잘한 미션 ──"));
		SetRow(Row_MinorSummary.Get(), TEXT("자잘한 미션 데이터 없음"));
		return;
	}

	const FMercenaryMissionProgress& MajorMission = SnapshotPtr->MissionState.MajorMission;

	if (Row_MajorMissionRows.IsValidIndex(0) == true)
	{
		SetRow(
			Row_MajorMissionRows[0].Get(),
			FString::Printf(
				TEXT("Mission: %s / Kind: %s"),
				*MajorMission.MissionId.ToString(),
				*UEnum::GetValueAsString(MajorMission.Kind)
			)
		);
	}

	if (Row_MajorMissionRows.IsValidIndex(1) == true)
	{
		SetRow(
			Row_MajorMissionRows[1].Get(),
			FString::Printf(
				TEXT("Target: %s / Faction: %s"),
				*MajorMission.TargetId.ToString(),
				*MajorMission.FactionId.ToString()
			)
		);
	}

	if (Row_MajorMissionRows.IsValidIndex(2) == true)
	{
		SetRow(
			Row_MajorMissionRows[2].Get(),
			FString::Printf(
				TEXT("Progress: %s / %d/%d"),
				*UEnum::GetValueAsString(MajorMission.ProgressState),
				MajorMission.CurrentValue,
				MajorMission.TargetValue
			)
		);
	}

	if (Row_MajorMissionRows.IsValidIndex(3) == true)
	{
		SetRow(
			Row_MajorMissionRows[3].Get(),
			FString::Printf(
				TEXT("RouteResult: %s"),
				*UEnum::GetValueAsString(MajorMission.RouteResultState)
			)
		);
	}

	if (Row_MajorMissionRows.IsValidIndex(4) == true)
	{
		SetRow(
			Row_MajorMissionRows[4].Get(),
			FString::Printf(
				TEXT("Created %d / Updated %d / Completed %d / Life %d"),
				MajorMission.CreatedTurnIndex,
				MajorMission.LastUpdatedTurnIndex,
				MajorMission.CompletedTurnIndex,
				MajorMission.ActiveLifeCycleTurnCount
			)
		);
	}

	SetRow(Row_MinorHeader.Get(), TEXT("── 자잘한 미션 ──"));

	const FMercenaryMissionState& MissionState = SnapshotPtr->MissionState;

	SetRow(
		Row_MinorSummary.Get(),
		FString::Printf(
			TEXT("활성 %d / 완료 %d / 총 생존주기 %d / 최대 %d"),
			MissionState.ActiveMinorMissionCount,
			MissionState.CompletedMinorMissionCount,
			MissionState.TotalMinorMissionLifeCycleTurnCount,
			MissionState.MaxMinorMissionLifeCycleTurnCount
		)
	);

	int32 VisibleIndex = 0;
	for (const FMercenaryMissionProgress& Mission : MissionState.MinorMissions)
	{
		if (Row_MinorMissionRows.IsValidIndex(VisibleIndex) == false)
		{
			break;
		}

		SetRow(
			Row_MinorMissionRows[VisibleIndex].Get(),
			FString::Printf(
				TEXT("%s / %s / %s / %d턴 / %d/%d"),
				*Mission.MissionId.ToString(),
				*UEnum::GetValueAsString(Mission.Kind),
				*UEnum::GetValueAsString(Mission.ProgressState),
				Mission.ActiveLifeCycleTurnCount,
				Mission.CurrentValue,
				Mission.TargetValue
			)
		);

		++VisibleIndex;
	}

	if (VisibleIndex == 0 && Row_MinorMissionRows.Num() > 0)
	{
		SetRow(Row_MinorMissionRows[0].Get(), TEXT("표시할 자잘한 미션 없음"));
	}
}

		}

		return;
	}

	int32 VisibleIndex = 0;
	for (const FMercenaryReturnConditionProgress& Condition : SnapshotPtr->ReturnConditions)
	{
		if (Condition.bVisibleInMissionInfo == false)

			continue;
		}

		{
			break;
		}

			Row_Conditions[VisibleIndex].Get(),
			FString::Printf(
				TEXT("%s: %d / %d  | Triggered=%s | Opportunities=%d"),
				*BuildReturnConditionKindLabel(Condition.Kind),

				Condition.bTriggered == true ? TEXT("true") : TEXT("false"),
				Condition.OpportunityCount
			)
		);

		++VisibleIndex;
	}

	if (VisibleIndex == 0 && Row_Conditions.Num() > 0)
	{
		SetRow(Row_Conditions[0].Get(), TEXT("표시할 돌아오는 조건 없음"));

				continue;
			}

			if (Index < UE_ARRAY_COUNT(GTabLabels))
			{
				Txt->SetText(PlayCoreDisplayText::UI(GTabLabels[Index]));

			{
				Txt->SetText(PlayCoreDisplayText::UI(TEXT("Unknown")));
			}

			Btn->SetContent(Txt);

			switch (Index)
			{
			case 0:

			case 1:

void UMercenaryInfoPanelWidget::PushDataToCurrentPanel()
{
	if (CurrentTab == EMercenaryInfoTab::Overall && Panel_Overall != nullptr && PartyDataPtr != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Overall->SetSessionData(
			*PartyDataPtr,
			*MercenaryDataPtr,
			AnalyticsDataPtr
		);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Skill && Panel_Skill != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Skill->SetSessionData(*MercenaryDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::PieceDeck && Panel_PieceDeck != nullptr && MercenaryDataPtr != nullptr && StageDeckPtr != nullptr)
	{
		Panel_PieceDeck->SetDeckData(
			MercenaryDataPtr->BasePieceTokenDeck,
			*StageDeckPtr
		);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Tree && Panel_Tree != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Tree->SetSessionData(*MercenaryDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Inventory && Panel_Inventory != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Inventory->SetSessionData(*MercenaryDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Stats && Panel_Stats != nullptr && PartyDataPtr != nullptr && MercenaryDataPtr != nullptr && AnalyticsDataPtr != nullptr)
	{
		Panel_Stats->SetSessionAnalytics(SessionAnalyticsPtr);
		Panel_Stats->SetSessionData(*PartyDataPtr, *MercenaryDataPtr, *AnalyticsDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Mission && Panel_Mission != nullptr)
	{
		Panel_Mission->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::ReturnCondition && Panel_ReturnCondition != nullptr)
	{
		Panel_ReturnCondition->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Return && Panel_Return != nullptr)
	{
		Panel_Return->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Dossier && Panel_Dossier != nullptr)
	{
		Panel_Dossier->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Region && Panel_Region != nullptr)
	{
		Panel_Region->SetRegionState(RegionStatePtr);
		Panel_Region->SetPartyRegionExpData(
			PartyRegionExpByIdPtr,
			PartyFactionExpByIdPtr,
			PartyTerrainExpByIdPtr,
			PartyFactionTerrainExpByIdPtr
		);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::EntityTest && Panel_EntityTest != nullptr)
	{
		Panel_EntityTest->SetDebugData(StageStatePtr, RegionStatePtr, DomainEntitiesPtr);
		return;
	}
}

FString UMercenaryInfoPanelWidget::BuildRegionExpText(const FMercenaryRegionRuntimeState* InRegionState) const
{
	if (InRegionState == nullptr || InRegionState->RegionExpById.Num() <= 0)
	{
		return TEXT("없음");
	}

	FString Result;

	for (const TPair<FName, int32>& Entry : InRegionState->RegionExpById)
	{
		if (Result.IsEmpty() == false)
		{
			Result += TEXT(" / ");
		}

		Result += FString::Printf(
			TEXT("%s:%d"),
			*Entry.Key.ToString(),
			Entry.Value
		);
	}

	return Result;
}

FString UMercenaryInfoPanelWidget::BuildStartRoadObjectText(const FMercenaryRegionRuntimeState* InRegionState) const
{
	if (InRegionState == nullptr || InRegionState->StartRoadObjectAcquiredById.Num() <= 0)
	{
		return TEXT("없음");
	}

	FString Result;

	for (const TPair<FName, bool>& Entry : InRegionState->StartRoadObjectAcquiredById)
	{
		if (Result.IsEmpty() == false)
		{
			Result += TEXT(" / ");
		}

		Result += FString::Printf(
			TEXT("%s:%s"),
			*Entry.Key.ToString(),
			Entry.Value == true ? TEXT("획득") : TEXT("미획득")
		);
	}

	return Result;
}

void UMercenaryInfoPanelWidget::RefreshTabButtons()
{
	const int32 TabCount = static_cast<int32>(EMercenaryInfoTab::Count);
	for (int32 Index = 0; Index < TabCount; ++Index)
	{
		if (TabButtons.IsValidIndex(Index) == false || TabButtons[Index] == nullptr)
		{
			continue;
		}

		const bool bActive = Index == static_cast<int32>(CurrentTab);

		const FLinearColor NormalColor = bActive == true
			? FLinearColor(0.20f, 0.45f, 0.80f, 1.0f)
			: FLinearColor(0.14f, 0.14f, 0.18f, 1.0f);

		FButtonStyle Style = TabButtons[Index]->GetStyle();
		Style.Normal.TintColor = FSlateColor(NormalColor);
		Style.Hovered.TintColor = FSlateColor(NormalColor * 1.1f);
		Style.Pressed.TintColor = FSlateColor(NormalColor * 0.9f);
		TabButtons[Index]->SetStyle(Style);

		if (TabTextBlocks.IsValidIndex(Index) == false || TabTextBlocks[Index] == nullptr)
		{
			continue;
		}

		FSlateFontInfo FontInfo = TabTextBlocks[Index]->GetFont();
		FontInfo.Size = 12;
		TabTextBlocks[Index]->SetFont(FontInfo);
		TabTextBlocks[Index]->SetColorAndOpacity(FSlateColor(FLinearColor::White));
	}
}

void UMercenaryInfoPanelWidget::RefreshPanelVisibility()
{
	struct FPanelEntry
	{
		UUserWidget* Widget = nullptr;
		EMercenaryInfoTab Tab = EMercenaryInfoTab::Overall;
	};

	const FPanelEntry Entries[] =
	{
		{ Panel_Overall.Get(), EMercenaryInfoTab::Overall },
		{ Panel_Skill.Get(), EMercenaryInfoTab::Skill },
		{ Panel_PieceDeck.Get(), EMercenaryInfoTab::PieceDeck },
		{ Panel_Tree.Get(), EMercenaryInfoTab::Tree },
		{ Panel_Inventory.Get(), EMercenaryInfoTab::Inventory },
		{ Panel_Stats.Get(), EMercenaryInfoTab::Stats },
		{ Panel_Mission.Get(), EMercenaryInfoTab::Mission },
		{ Panel_ReturnCondition.Get(), EMercenaryInfoTab::ReturnCondition },
		{ Panel_Return.Get(), EMercenaryInfoTab::Return },
		{ Panel_Dossier.Get(), EMercenaryInfoTab::Dossier },
		{ Panel_Region.Get(), EMercenaryInfoTab::Region },
		{ Panel_EntityTest.Get(), EMercenaryInfoTab::EntityTest },
	};

	for (const FPanelEntry& Entry : Entries)
	{
		if (Entry.Widget == nullptr)
		{
			continue;
		}

		Entry.Widget->SetVisibility(
			Entry.Tab == CurrentTab
				? ESlateVisibility::SelfHitTestInvisible
				: ESlateVisibility::Collapsed
		);
	}
}
