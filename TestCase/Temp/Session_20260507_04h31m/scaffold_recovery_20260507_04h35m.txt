UTextBlock* UMercenaryInfoRegionWidget::AddRow(UVerticalBox* InBox, const FLinearColor& InColor)
{
	if (InBox == nullptr || WidgetTree == nullptr)
	{
		return nullptr;
	}

	UTextBlock* Block = WidgetTree->ConstructWidget<UTextBlock>(UTextBlock::StaticClass());
	if (Block == nullptr)
	{
		return nullptr;
	}

	FSlateFontInfo FontInfo = Block->GetFont();
	FontInfo.Size = 14;
	Block->SetFont(FontInfo);
	Block->SetColorAndOpacity(FSlateColor(InColor));
	Block->SetAutoWrapText(true);

	if (UVerticalBoxSlot* TempSlot = InBox->AddChildToVerticalBox(Block))
	{
		TempSlot->SetPadding(FMargin(0.0f, 0.0f, 0.0f, 4.0f));
	}

	return Block;
}

FString UMercenaryInfoRegionWidget::BuildRegionExpText() const
{
	const FName FactionKey = CachedRegionState.CurrentFactionId.IsNone() == true
		? FName(TEXT("Faction.None"))
		: CachedRegionState.CurrentFactionId;

	const FName TerrainKey = FName(*FString::Printf(
		TEXT("Terrain.%s"),
		*UEnum::GetValueAsString(CachedRegionState.CurrentRegionKind)
	));

	const FName FactionTerrainKey = FName(*FString::Printf(
		TEXT("%s.%s"),
		*FactionKey.ToString(),
		*TerrainKey.ToString()
	));

	FString Result = FString::Printf(
		TEXT("현재 3축 경험치\n- Faction %s: %d\n- Terrain %s: %d\n- FactionTerrain %s: %d"),
		*FactionKey.ToString(),
		CachedRegionState.FactionExpById.FindRef(FactionKey),
		*TerrainKey.ToString(),
		CachedRegionState.TerrainExpById.FindRef(TerrainKey),
		*FactionTerrainKey.ToString(),
		CachedRegionState.FactionTerrainExpById.FindRef(FactionTerrainKey)
	);

	Result += TEXT("\n\n리전별 누적");

	if (CachedRegionState.RegionExpById.Num() <= 0)
	{
		Result += TEXT("\n- 없음");
	}
	else
	{
		for (const TPair<FName, int32>& Entry : CachedRegionState.RegionExpById)
		{
			Result += FString::Printf(TEXT("\n- %s: %d"), *Entry.Key.ToString(), Entry.Value);
		}
	}

	Result += TEXT("\n\n팩션별 누적");

	if (CachedRegionState.FactionExpById.Num() <= 0)
	{
		Result += TEXT("\n- 없음");
	}
	else
	{
		for (const TPair<FName, int32>& Entry : CachedRegionState.FactionExpById)
		{
			Result += FString::Printf(TEXT("\n- %s: %d"), *Entry.Key.ToString(), Entry.Value);
		}
	}

	Result += TEXT("\n\n지형별 누적");

	if (CachedRegionState.TerrainExpById.Num() <= 0)
	{
		Result += TEXT("\n- 없음");
	}
	else
	{
		for (const TPair<FName, int32>& Entry : CachedRegionState.TerrainExpById)
		{
			Result += FString::Printf(TEXT("\n- %s: %d"), *Entry.Key.ToString(), Entry.Value);
		}
	}

	Result += TEXT("\n\n팩션+지형별 누적");

	if (CachedRegionState.FactionTerrainExpById.Num() <= 0)
	{
		Result += TEXT("\n- 없음");
	}
	else
	{
		for (const TPair<FName, int32>& Entry : CachedRegionState.FactionTerrainExpById)
		{
			Result += FString::Printf(TEXT("\n- %s: %d"), *Entry.Key.ToString(), Entry.Value);
		}
	}

	return Result;
}

FString UMercenaryInfoRegionWidget::BuildPartyRegionExpText() const
{
	auto AppendMap = [](FString& InOutText, const TCHAR* InLabel, const TMap<FName, int32>& InMap)
	{
		InOutText += FString::Printf(TEXT("%s"), InLabel);

		if (InMap.Num() <= 0)
		{
			InOutText += TEXT("\n- 없음\n");
			return;
		}

		for (const TPair<FName, int32>& Entry : InMap)
		{
			InOutText += FString::Printf(TEXT("\n- %s: %d"), *Entry.Key.ToString(), Entry.Value);
		}

		InOutText += TEXT("\n");
	};

	FString Result;
	AppendMap(Result, TEXT("Party Region"), CachedPartyRegionExpById);
	Result += TEXT("\n");
	AppendMap(Result, TEXT("Party Faction"), CachedPartyFactionExpById);
	Result += TEXT("\n");
	AppendMap(Result, TEXT("Party Terrain"), CachedPartyTerrainExpById);
	Result += TEXT("\n");
	AppendMap(Result, TEXT("Party Faction+Terrain"), CachedPartyFactionTerrainExpById);

	return Result;
}
FString UMercenaryInfoRegionWidget::BuildRegionMasteryText() const
{
	const int32 Exp = CachedRegionState.CurrentRegionExp;

	if (Exp >= 100)
	{
		return TEXT("숙련도: 익숙함 이상 / 이 리전의 흐름을 꽤 파악한 상태");
	}

	if (Exp >= 50)
	{
		return TEXT("숙련도: 익숙해지는 중 / 주요 위험과 보상을 조금 파악한 상태");
	}

	if (Exp >= 20)
	{
		return TEXT("숙련도: 초입 / 아직 리전 정보가 제한적인 상태");
	}

	return TEXT("숙련도: 미숙 / 리전 정보 거의 없음");
}

FString UMercenaryInfoRegionWidget::BuildTravelPathStateText() const
{
	const FMercenaryTravelPathStateProfile_Table& PathProfile = CachedRegionState.CurrentRegionDef.TravelPathStateProfile;

	FString Result = FString::Printf(
		TEXT("길 상태 테이블: Enabled=%s / RollChance=%d"),
		PathProfile.bEnabled == true ? TEXT("true") : TEXT("false"),
		PathProfile.PathStateRollChance
	);

	if (PathProfile.PossiblePathStates.Num() <= 0)
	{
		Result += TEXT(" / Possible=없음");
		return Result;
	}

	Result += TEXT(" / Possible=");

	for (int32 Index = 0; Index < PathProfile.PossiblePathStates.Num(); ++Index)
	{
		if (Index > 0)
		{
			Result += TEXT(", ");
		}

		Result += PathProfile.PossiblePathStates[Index].ToString();
	}

	return Result;
}

FString UMercenaryInfoRegionWidget::BuildStartRoadObjectDefText() const
{
	if (CachedRegionState.CurrentRegionDef.StartRoadObjectIds.Num() <= 0)
	{
		return TEXT("시작의 길 오브젝트 테이블: 없음");
	}

	FString Result = TEXT("시작의 길 오브젝트 테이블: ");

	for (int32 Index = 0; Index < CachedRegionState.CurrentRegionDef.StartRoadObjectIds.Num(); ++Index)
	{
		if (Index > 0)
		{
			Result += TEXT(", ");
		}

		Result += CachedRegionState.CurrentRegionDef.StartRoadObjectIds[Index].ToString();
	}

	return Result;
}

FString UMercenaryInfoRegionWidget::BuildStartRoadObjectText() const
{
	if (CachedRegionState.StartRoadObjectAcquiredById.Num() <= 0)
	{
		return TEXT("없음");
	}

	FString Result;

	for (const TPair<FName, bool>& Entry : CachedRegionState.StartRoadObjectAcquiredById)
	{
		if (Result.IsEmpty() == false)
		{
			Result += TEXT("\n");
		}

		Result += FString::Printf(
			TEXT("- %s: %s"),
			*Entry.Key.ToString(),
			Entry.Value == true ? TEXT("획득") : TEXT("미획득")
		);
	}

	return Result;
}

void UMercenaryInfoEntityTestWidget::RefreshBoard()
{
	if (WidgetTree == nullptr || BoardCanvas == nullptr)
	{
		return;
	}

	BoardCanvas->ClearChildren();

	const int32 StageWidth = FMath::Max(1, CachedStageState.StageWidth);
	const int32 StageHeight = FMath::Max(1, CachedStageState.StageHeight);

		BoardSizeBox->SetHeightOverride(static_cast<float>(StageHeight) * GStageCellSize);
	}

	for (int32 StageY = 0; StageY < StageHeight; ++StageY)

		for (int32 StageX = 0; StageX < StageWidth; ++StageX)

			AddBoardCell(StageX, StageY);

	}
}

void UMercenaryInfoEntityTestWidget::AddBoardCell(const int32 InStageX, const int32 InStageY)
{
	if (WidgetTree == nullptr || BoardCanvas == nullptr)
	{
		return;
	}

	bool bInnerBoard = CachedStageState.IsInnerBoardStageCoord(InStageX, InStageY);
	bool bBlocked = false;

	for (const FMercenaryStageCellState& Cell : CachedStageState.Cells)
	{
		if (Cell.StageX == InStageX && Cell.StageY == InStageY)
		{
			bInnerBoard = Cell.bInnerBoard;
			bBlocked = Cell.bBlocked;
			break;
		}
	}

	const bool bHovered = InStageX == HoverStageX && InStageY == HoverStageY;
	const bool bHasEntity = HasLivingEntityAtStageCoord(InStageX, InStageY);

	FLinearColor CellColor = bInnerBoard == true ? GInnerCellColor : GOuterCellColor;

	if (bBlocked == true)
	{
		CellColor = GBlockedCellColor;
	}

	if (bHasEntity == true)
	{
		CellColor = GEntityCellColor;
	}

	if (bHovered == true)
	{
		CellColor = GHoverCellColor;
	}

	UBorder* CellBorder = WidgetTree->ConstructWidget<UBorder>(
		UBorder::StaticClass(),
		*FString::Printf(TEXT("StageCell_%d_%d"), InStageX, InStageY)
	);

	if (CellBorder == nullptr)
	{
		return;
	}

	CellBorder->SetBrushColor(CellColor);
	CellBorder->SetPadding(FMargin(1.0f));

	if (UCanvasPanelSlot* CellSlot = BoardCanvas->AddChildToCanvas(CellBorder))
	{
		CellSlot->SetAnchors(FAnchors(0.0f, 0.0f));
		CellSlot->SetAlignment(FVector2D(0.0f, 0.0f));
		CellSlot->SetPosition(FVector2D(
			static_cast<float>(InStageX) * GStageCellSize,
			static_cast<float>(InStageY) * GStageCellSize
		));
		CellSlot->SetSize(FVector2D(GStageCellSize - 1.0f, GStageCellSize - 1.0f));
		CellSlot->SetAutoSize(false);
		CellSlot->SetZOrder(bHovered == true ? 10 : 0);
	}
}

bool UMercenaryInfoEntityTestWidget::HasLivingEntityAtStageCoord(const int32 InStageX, const int32 InStageY) const
{
	for (const FPlayCoreDomainEntityState& EntityState : CachedEntities)
	{
		if (EntityState.bAlive == false)
		{
			continue;
		}

		const int32 EntityStageX = CachedStageState.BoardOffsetX + EntityState.Coord.X;
		const int32 EntityStageY = CachedStageState.BoardOffsetY + EntityState.Coord.Y;

		if (EntityStageX == InStageX && EntityStageY == InStageY)

			return true;
		}
	}

	return false;
}

FString UMercenaryInfoEntityTestWidget::BuildHoveredEntityText() const
{
	if (HoverStageX == INDEX_NONE || HoverStageY == INDEX_NONE)
	{
		return TEXT("Hover: 없음");
	}

	FString Result = FString::Printf(TEXT("Hover StageCoord=(%d,%d)"), HoverStageX, HoverStageY);

	if (CachedStageState.IsInnerBoardStageCoord(HoverStageX, HoverStageY) == true)
	{
		FPlayCoreBoardCoord BoardCoord;
		CachedStageState.TryConvertStageCoordToBoardCoord(HoverStageX, HoverStageY, BoardCoord);
		Result += FString::Printf(TEXT(" / BoardCoord=(%d,%d)"), BoardCoord.X, BoardCoord.Y);
	}

	bool bFound = false;

	for (const FPlayCoreDomainEntityState& EntityState : CachedEntities)
	{
		const int32 EntityStageX = CachedStageState.BoardOffsetX + EntityState.Coord.X;
		const int32 EntityStageY = CachedStageState.BoardOffsetY + EntityState.Coord.Y;

		if (EntityStageX != HoverStageX || EntityStageY != HoverStageY)
		{
			continue;
		}

		bFound = true;

		Result += FString::Printf(
			TEXT("\n- EntityId=%s / Type=%s / Outcome=%s / HP=%d / Durability=%d / Alive=%s"),
			*EntityState.EntityId.ToString(),
			*UEnum::GetValueAsString(EntityState.EntityType),
			*UEnum::GetValueAsString(EntityState.OutcomeKind),
			EntityState.Hp,
			EntityState.DurabilityCount,
			EntityState.bAlive == true ? TEXT("true") : TEXT("false")
		);
	}

	if (bFound == false)
	{
		Result += TEXT("\n- Entity: 없음");
	}

	return Result;
}

FString UMercenaryInfoEntityTestWidget::BuildEntityListText() const
{
	if (CachedEntities.Num() <= 0)
	{
		return TEXT("── 엔티티 목록 ──\n없음");
	}

	int32 AliveCount = 0;
	int32 DeadCount = 0;

	for (const FPlayCoreDomainEntityState& EntityState : CachedEntities)
	{
		if (EntityState.bAlive == true)
		{
			++AliveCount;
		}
		else
		{
			++DeadCount;
		}
	}

	FString Result = FString::Printf(
		TEXT("── 엔티티 목록 ──\nTotal=%d / Alive=%d / Dead=%d"),
		CachedEntities.Num(),
		AliveCount,
		DeadCount
	);

	for (const FPlayCoreDomainEntityState& EntityState : CachedEntities)
	{
		const int32 StageX = CachedStageState.BoardOffsetX + EntityState.Coord.X;
		const int32 StageY = CachedStageState.BoardOffsetY + EntityState.Coord.Y;

		Result += FString::Printf(
			TEXT("\n- %s / Type=%s / Board=(%d,%d) / Stage=(%d,%d) / HP=%d / Dur=%d / Alive=%s / Outcome=%s"),
			*EntityState.EntityId.ToString(),
			*UEnum::GetValueAsString(EntityState.EntityType),
			EntityState.Coord.X,
			EntityState.Coord.Y,
			StageX,
			StageY,
			EntityState.Hp,
			EntityState.DurabilityCount,
			EntityState.bAlive == true ? TEXT("true") : TEXT("false"),
			*UEnum::GetValueAsString(EntityState.OutcomeKind)
		);
	}

	return Result;
}

void UMercenaryInfoPanelWidget::BuildWidgetTree()
{
	if (WidgetTree == nullptr)
	{
		return;
	}

	TabButtons.Reset();
	TabTextBlocks.Reset();

	RootSizeBox = WidgetTree->ConstructWidget<USizeBox>(USizeBox::StaticClass(), TEXT("RootSizeBox"));
	if (RootSizeBox == nullptr)
	{
		return;
	}

	RootSizeBox->SetWidthOverride(GPanelWidth);
	RootSizeBox->SetHeightOverride(GPanelHeight);
	WidgetTree->RootWidget = RootSizeBox;

	RootBorder = WidgetTree->ConstructWidget<UBorder>(UBorder::StaticClass(), TEXT("RootBorder"));
	if (RootBorder == nullptr)
	{
		return;
	}

	RootBorder->SetBrushColor(FLinearColor(0.025f, 0.025f, 0.035f, 0.96f));
	RootBorder->SetPadding(FMargin(12.0f));
	RootSizeBox->SetContent(RootBorder);

	RootVerticalBox = WidgetTree->ConstructWidget<UVerticalBox>(UVerticalBox::StaticClass(), TEXT("RootVerticalBox"));
	if (RootVerticalBox == nullptr)
	{
		return;
	}

	RootBorder->SetContent(RootVerticalBox);

	TabButtonRow = WidgetTree->ConstructWidget<UHorizontalBox>(UHorizontalBox::StaticClass(), TEXT("TabButtonRow"));
	if (TabButtonRow != nullptr)
	{
		if (UVerticalBoxSlot* TabRowSlot = RootVerticalBox->AddChildToVerticalBox(TabButtonRow))
		{
			TabRowSlot->SetPadding(FMargin(0.0f, 0.0f, 0.0f, 8.0f));
			TabRowSlot->SetSize(FSlateChildSize(ESlateSizeRule::Automatic));
		}

		const int32 TabCount = static_cast<int32>(EMercenaryInfoTab::Count);
		for (int32 Index = 0; Index < TabCount; ++Index)
		{
			UButton* Btn = WidgetTree->ConstructWidget<UButton>(
				UButton::StaticClass(),
				*FString::Printf(TEXT("TabBtn_%d"), Index)
			);

			UTextBlock* Txt = WidgetTree->ConstructWidget<UTextBlock>(
				UTextBlock::StaticClass(),
				*FString::Printf(TEXT("TabTxt_%d"), Index)
			);

			if (Btn == nullptr || Txt == nullptr)
			{
				continue;
			}

			if (Index < UE_ARRAY_COUNT(GTabLabels))
			{
				Txt->SetText(PlayCoreDisplayText::UI(GTabLabels[Index]));
			}
			else
			{
				Txt->SetText(PlayCoreDisplayText::UI(TEXT("Unknown")));
			}

			Btn->SetContent(Txt);

			switch (Index)
			{
			case 0:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabOverall);
				break;

			case 1:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabSkill);
				break;

			case 2:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabPieceDeck);
				break;

			case 3:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabTree);
				break;

			case 4:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabInventory);
				break;

			case 5:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabStats);
				break;

			case 6:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabMission);
				break;

			case 7:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabReturnCondition);
				break;

			case 8:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabReturn);
				break;

			case 9:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabDossier);
				break;

			case 10:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabRegion);
				break;

			case 11:
				Btn->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleTabEntityTest);
				break;

			default:
				break;
			}

			if (UHorizontalBoxSlot* BtnSlot = TabButtonRow->AddChildToHorizontalBox(Btn))
			{
				BtnSlot->SetPadding(FMargin(0.0f, 0.0f, GTabButtonSpacing, 0.0f));
				BtnSlot->SetSize(FSlateChildSize(ESlateSizeRule::Fill));
			}

			TabButtons.Add(Btn);
			TabTextBlocks.Add(Txt);
		}
	}

	Panel_Overall = WidgetTree->ConstructWidget<UMercenaryInfoOverallWidget>(UMercenaryInfoOverallWidget::StaticClass(), TEXT("Panel_Overall"));
	Panel_Skill = WidgetTree->ConstructWidget<UMercenaryInfoSkillWidget>(UMercenaryInfoSkillWidget::StaticClass(), TEXT("Panel_Skill"));
	Panel_PieceDeck = WidgetTree->ConstructWidget<UMercenaryInfoPieceDeckWidget>(UMercenaryInfoPieceDeckWidget::StaticClass(), TEXT("Panel_PieceDeck"));
	Panel_Tree = WidgetTree->ConstructWidget<UMercenaryInfoTreeWidget>(UMercenaryInfoTreeWidget::StaticClass(), TEXT("Panel_Tree"));
	Panel_Inventory = WidgetTree->ConstructWidget<UMercenaryInfoInventoryWidget>(UMercenaryInfoInventoryWidget::StaticClass(), TEXT("Panel_Inventory"));
	Panel_Stats = WidgetTree->ConstructWidget<UMercenaryInfoStatsWidget>(UMercenaryInfoStatsWidget::StaticClass(), TEXT("Panel_Stats"));
	Panel_Mission = WidgetTree->ConstructWidget<UMercenaryInfoMissionWidget>(UMercenaryInfoMissionWidget::StaticClass(), TEXT("Panel_Mission"));
	Panel_ReturnCondition = WidgetTree->ConstructWidget<UMercenaryInfoReturnConditionWidget>(UMercenaryInfoReturnConditionWidget::StaticClass(), TEXT("Panel_ReturnCondition"));
	Panel_Return = WidgetTree->ConstructWidget<UMercenaryInfoReturnWidget>(UMercenaryInfoReturnWidget::StaticClass(), TEXT("Panel_Return"));
	Panel_Dossier = WidgetTree->ConstructWidget<UMercenaryInfoDossierWidget>(UMercenaryInfoDossierWidget::StaticClass(), TEXT("Panel_Dossier"));
	Panel_Region = WidgetTree->ConstructWidget<UMercenaryInfoRegionWidget>(UMercenaryInfoRegionWidget::StaticClass(), TEXT("Panel_Region"));
	Panel_EntityTest = WidgetTree->ConstructWidget<UMercenaryInfoEntityTestWidget>(UMercenaryInfoEntityTestWidget::StaticClass(), TEXT("Panel_EntityTest"));

	if (Panel_Return != nullptr)
	{
		Panel_Return->OnDebugReturnRequested.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleDebugReturnClicked);
	}

	UUserWidget* AllPanels[] =
	{
		Panel_Overall.Get(),
		Panel_Skill.Get(),
		Panel_PieceDeck.Get(),
		Panel_Tree.Get(),
		Panel_Inventory.Get(),
		Panel_Stats.Get(),
		Panel_Mission.Get(),
		Panel_ReturnCondition.Get(),
		Panel_Return.Get(),
		Panel_Dossier.Get(),
		Panel_Region.Get(),
		Panel_EntityTest.Get(),
	};

	for (UUserWidget* Panel : AllPanels)
	{
		if (Panel == nullptr)
		{
			continue;
		}

		if (UVerticalBoxSlot* PanelSlot = RootVerticalBox->AddChildToVerticalBox(Panel))
		{
			PanelSlot->SetPadding(FMargin(0.0f));
			PanelSlot->SetSize(FSlateChildSize(ESlateSizeRule::Fill));
		}
	}

	CloseButton = WidgetTree->ConstructWidget<UButton>(UButton::StaticClass(), TEXT("CloseButton"));
	CloseTextBlock = WidgetTree->ConstructWidget<UTextBlock>(UTextBlock::StaticClass(), TEXT("CloseTextBlock"));

	if (CloseButton != nullptr && CloseTextBlock != nullptr)
	{
		CloseTextBlock->SetText(PlayCoreDisplayText::UI(TEXT("닫기 (ESC)")));
		CloseButton->SetContent(CloseTextBlock);
		CloseButton->OnClicked.AddDynamic(this, &UMercenaryInfoPanelWidget::HandleCloseClicked);

		if (UVerticalBoxSlot* CloseSlot = RootVerticalBox->AddChildToVerticalBox(CloseButton))
		{
			CloseSlot->SetPadding(FMargin(12.0f, 6.0f, 12.0f, 10.0f));
			CloseSlot->SetHorizontalAlignment(HAlign_Right);
			CloseSlot->SetSize(FSlateChildSize(ESlateSizeRule::Automatic));
		}
	}
}

void UMercenaryInfoPanelWidget::PushDataToCurrentPanel()
{
	if (CurrentTab == EMercenaryInfoTab::Overall && Panel_Overall != nullptr && PartyDataPtr != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Overall->SetSessionData(
			*PartyDataPtr,
			*MercenaryDataPtr,
			AnalyticsDataPtr
		);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Skill && Panel_Skill != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Skill->SetSessionData(*MercenaryDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::PieceDeck && Panel_PieceDeck != nullptr && MercenaryDataPtr != nullptr && StageDeckPtr != nullptr)
	{
		Panel_PieceDeck->SetDeckData(
			MercenaryDataPtr->BasePieceTokenDeck,
			*StageDeckPtr
		);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Tree && Panel_Tree != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Tree->SetSessionData(*MercenaryDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Inventory && Panel_Inventory != nullptr && MercenaryDataPtr != nullptr)
	{
		Panel_Inventory->SetSessionData(*MercenaryDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Stats && Panel_Stats != nullptr && PartyDataPtr != nullptr && MercenaryDataPtr != nullptr && AnalyticsDataPtr != nullptr)
	{
		Panel_Stats->SetSessionAnalytics(SessionAnalyticsPtr);
		Panel_Stats->SetSessionData(*PartyDataPtr, *MercenaryDataPtr, *AnalyticsDataPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Mission && Panel_Mission != nullptr)
	{
		Panel_Mission->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::ReturnCondition && Panel_ReturnCondition != nullptr)
	{
		Panel_ReturnCondition->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Return && Panel_Return != nullptr)
	{
		Panel_Return->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Dossier && Panel_Dossier != nullptr)
	{
		Panel_Dossier->SetSnapshot(RaidResultSnapshotPtr);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::Region && Panel_Region != nullptr)
	{
		Panel_Region->SetRegionState(RegionStatePtr);
		Panel_Region->SetPartyRegionExpData(
			PartyRegionExpByIdPtr,
			PartyFactionExpByIdPtr,
			PartyTerrainExpByIdPtr,
			PartyFactionTerrainExpByIdPtr
		);
		return;
	}

	if (CurrentTab == EMercenaryInfoTab::EntityTest && Panel_EntityTest != nullptr)
	{
		Panel_EntityTest->SetDebugData(StageStatePtr, RegionStatePtr, DomainEntitiesPtr);
		return;
	}
}

FString UMercenaryInfoPanelWidget::BuildRegionExpText(const FMercenaryRegionRuntimeState* InRegionState) const
{
	if (InRegionState == nullptr || InRegionState->RegionExpById.Num() <= 0)
	{
		return TEXT("없음");
	}

	FString Result;

	for (const TPair<FName, int32>& Entry : InRegionState->RegionExpById)
	{
		if (Result.IsEmpty() == false)
		{
			Result += TEXT(" / ");
		}

		Result += FString::Printf(
			TEXT("%s:%d"),
			*Entry.Key.ToString(),
			Entry.Value
		);
	}

	return Result;
}

FString UMercenaryInfoPanelWidget::BuildStartRoadObjectText(const FMercenaryRegionRuntimeState* InRegionState) const
{
	if (InRegionState == nullptr || InRegionState->StartRoadObjectAcquiredById.Num() <= 0)
	{
		return TEXT("없음");
	}

	FString Result;

	for (const TPair<FName, bool>& Entry : InRegionState->StartRoadObjectAcquiredById)
	{
		if (Result.IsEmpty() == false)
		{
			Result += TEXT(" / ");
		}

		Result += FString::Printf(
			TEXT("%s:%s"),
			*Entry.Key.ToString(),
			Entry.Value == true ? TEXT("획득") : TEXT("미획득")
		);
	}

	return Result;
}

void UMercenaryInfoPanelWidget::RefreshTabButtons()
{
	const int32 TabCount = static_cast<int32>(EMercenaryInfoTab::Count);
	for (int32 Index = 0; Index < TabCount; ++Index)
	{
		if (TabButtons.IsValidIndex(Index) == false || TabButtons[Index] == nullptr)
		{
			continue;
		}

		const bool bActive = Index == static_cast<int32>(CurrentTab);

		const FLinearColor NormalColor = bActive == true
			? FLinearColor(0.20f, 0.45f, 0.80f, 1.0f)
			: FLinearColor(0.14f, 0.14f, 0.18f, 1.0f);

		FButtonStyle Style = TabButtons[Index]->GetStyle();
		Style.Normal.TintColor = FSlateColor(NormalColor);
		Style.Hovered.TintColor = FSlateColor(NormalColor * 1.1f);
		Style.Pressed.TintColor = FSlateColor(NormalColor * 0.9f);
		TabButtons[Index]->SetStyle(Style);

		if (TabTextBlocks.IsValidIndex(Index) == false || TabTextBlocks[Index] == nullptr)
		{
			continue;
		}

		FSlateFontInfo FontInfo = TabTextBlocks[Index]->GetFont();
		FontInfo.Size = 12;
		TabTextBlocks[Index]->SetFont(FontInfo);
		TabTextBlocks[Index]->SetColorAndOpacity(FSlateColor(FLinearColor::White));
	}
}

void UMercenaryInfoPanelWidget::RefreshPanelVisibility()
{
	struct FPanelEntry
	{
		UUserWidget* Widget = nullptr;
		EMercenaryInfoTab Tab = EMercenaryInfoTab::Overall;
	};

	const FPanelEntry Entries[] =
	{
		{ Panel_Overall.Get(), EMercenaryInfoTab::Overall },
		{ Panel_Skill.Get(), EMercenaryInfoTab::Skill },
		{ Panel_PieceDeck.Get(), EMercenaryInfoTab::PieceDeck },
		{ Panel_Tree.Get(), EMercenaryInfoTab::Tree },
		{ Panel_Inventory.Get(), EMercenaryInfoTab::Inventory },
		{ Panel_Stats.Get(), EMercenaryInfoTab::Stats },
		{ Panel_Mission.Get(), EMercenaryInfoTab::Mission },
		{ Panel_ReturnCondition.Get(), EMercenaryInfoTab::ReturnCondition },
		{ Panel_Return.Get(), EMercenaryInfoTab::Return },
		{ Panel_Dossier.Get(), EMercenaryInfoTab::Dossier },
		{ Panel_Region.Get(), EMercenaryInfoTab::Region },
		{ Panel_EntityTest.Get(), EMercenaryInfoTab::EntityTest },
	};

	for (const FPanelEntry& Entry : Entries)
	{
		if (Entry.Widget == nullptr)
		{
			continue;
		}

		Entry.Widget->SetVisibility(
			Entry.Tab == CurrentTab
