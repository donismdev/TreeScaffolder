void AMercenaryEntryGameMode::EndPlay(const EEndPlayReason::Type EndPlayReason)
{
	if (ActiveFlow != nullptr)
	{
		// TODO: Replace with real result extraction from ActiveFlow
		CDR_Contracts::FMercenaryRunResult Result;
		Result.bCleared = false;
		Result.Score = 0;
		Result.EarnedGold = 0;
		Result.Seed = 0;
		Result.PlaySeconds = 0.0f;

		UWorld* const WorldPtr = GetWorld();
		if (WorldPtr != nullptr)
		{
			UGameInstance* const GameInstancePtr = WorldPtr->GetGameInstance();
			if (GameInstancePtr != nullptr)
			{
				CDR_Contracts::IModeContractsProvider* const ProviderPtr = Cast<CDR_Contracts::IModeContractsProvider>(GameInstancePtr);
				if (ProviderPtr != nullptr)
				{
					CDR_Contracts::IMercenaryModeContract* const ContractPtr = ProviderPtr->GetMercenaryModeContract();
					if (ContractPtr != nullptr)
					{
						ContractPtr->SetLastResult(Result);
					}
				}
			}
		}

		ActiveFlow->Shutdown();
		ActiveFlow.Reset();
	}

	Super::EndPlay(EndPlayReason);
}

void AMercenaryEntryGameMode::InitializeFromStartContext()
{
	if (bInitializedFromStartContext == true)
	{
		return;
	}

	UWorld* const WorldPtr = GetWorld();
	if (WorldPtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MercenaryEntry] InitializeFromStartContext: World is null"));
		return;
	}

	UGameInstance* const GameInstancePtr = WorldPtr->GetGameInstance();
	if (GameInstancePtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MercenaryEntry] InitializeFromStartContext: GameInstance is null"));
		return;
	}

	CDR_Contracts::IModeContractsProvider* const ProviderPtr = Cast<CDR_Contracts::IModeContractsProvider>(GameInstancePtr);
	if (ProviderPtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MercenaryEntry] InitializeFromStartContext: GameInstance does not implement IModeContractsProvider"));
		return;
	}

	CDR_Contracts::IMercenaryModeContract* const ContractPtr = ProviderPtr->GetMercenaryModeContract();
	if (ContractPtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MercenaryEntry] InitializeFromStartContext: Mercenary contract is null"));
		return;
	}

	CDR_Contracts::FMercenaryStartContext StartContext;
	const CDR_Contracts::EConsumeResult ConsumeResult = ContractPtr->ConsumeStartContext(StartContext);
	if (ConsumeResult != CDR_Contracts::EConsumeResult::Success)
	{
		UE_LOG(LogTemp, Error, TEXT("[MercenaryEntry] InitializeFromStartContext: StartContext missing"));
		return;
	}

	ActiveFlow = MakeUnique<CDR_ModeMercenary::FModeMercenaryFlow>();
	if (ActiveFlow == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MercenaryEntry] InitializeFromStartContext: Failed to allocate ModeMercenaryFlow"));
		return;
	}

	ActiveFlow->Initialize(WorldPtr);
	bInitializedFromStartContext = true;
}

@@@FILE_BEGIN {{Root}}/Bootstrap/MainHubGameMode.cpp
#include "Bootstrap/MainHubGameMode.h"

#include "Bootstrap/MainHubPlayerController.h"
#include "UIRoot/MainMenuWidget.h"

#include "MercenaryModeContract.h"
#include "ModeContractsProvider.h"
#include "ScoreModeContract.h"

#include "Engine/World.h"
#include "GameFramework/PlayerController.h"
#include "Kismet/GameplayStatics.h"

namespace
{
	static const FName NAME_Mode_Mercenary(TEXT("Mercenary"));
	static const FName NAME_Mode_Score(TEXT("Score"));

	static const FName NAME_Level_MercenaryEntry(TEXT("MercenaryEntry"));
	static const FName NAME_Level_ScoreEntry(TEXT("ScoreEntry"));
}

AMainHubGameMode::AMainHubGameMode()
{
	PlayerControllerClass = AMainHubPlayerController::StaticClass();
}

void AMainHubGameMode::BeginPlay()
{
	Super::BeginPlay();

	UWorld* const WorldPtr = GetWorld();
	if (WorldPtr == nullptr)
	{
		return;
	}

	ConsumeModeResultsIfAny(WorldPtr);

	APlayerController* const PlayerControllerPtr = WorldPtr->GetFirstPlayerController();
	if (PlayerControllerPtr == nullptr)
	{
		return;
	}

	MainMenuWidget = CreateWidget<UMainMenuWidget>(PlayerControllerPtr, UMainMenuWidget::StaticClass());
	if (MainMenuWidget == nullptr)
	{
		return;
	}

	MainMenuWidget->OnModeSelected.AddUObject(this, &AMainHubGameMode::HandleModeSelected);
	MainMenuWidget->AddToViewport(0);

	PlayerControllerPtr->SetInputMode(FInputModeUIOnly());
	PlayerControllerPtr->bShowMouseCursor = true;
}

void AMainHubGameMode::HandleModeSelected(const FName SelectedModeIdValue)
{
	UWorld* const WorldPtr = GetWorld();
	if (WorldPtr == nullptr)
	{
		return;
	}

	if (TryEnterMercenaryMode(WorldPtr, SelectedModeIdValue) == true)
	{
		return;
	}

	if (TryEnterScoreMode(WorldPtr, SelectedModeIdValue) == true)
	{
		return;
	}

	UE_LOG(LogTemp, Error, TEXT("[MainHub] Unknown mode id: %s"), *SelectedModeIdValue.ToString());
}

void AMainHubGameMode::ConsumeModeResultsIfAny(UWorld* const WorldPtr)
{
	if (WorldPtr == nullptr)
	{
		return;
	}

	UGameInstance* const GameInstancePtr = WorldPtr->GetGameInstance();
	if (GameInstancePtr == nullptr)
	{
		return;
	}

	CDR_Contracts::IModeContractsProvider* const ProviderPtr = Cast<CDR_Contracts::IModeContractsProvider>(GameInstancePtr);
	if (ProviderPtr == nullptr)
	{
		UE_LOG(LogTemp, Warning, TEXT("[MainHub] ConsumeModeResultsIfAny: GameInstance does not implement IModeContractsProvider"));
		return;
	}

	{
		CDR_Contracts::IMercenaryModeContract* const MercenaryContractPtr = ProviderPtr->GetMercenaryModeContract();
		if (MercenaryContractPtr != nullptr)
		{
			CDR_Contracts::FMercenaryRunResult MercenaryResult;
			const CDR_Contracts::EConsumeResult ConsumeResult = MercenaryContractPtr->ConsumeLastResult(MercenaryResult);
			if (ConsumeResult == CDR_Contracts::EConsumeResult::Success)
			{
				UE_LOG(LogTemp, Log, TEXT("[MainHub] Consumed Mercenary result: Cleared=%s Score=%d Gold=%d Seed=%d Seconds=%.2f"),
					(MercenaryResult.bCleared == true) ? TEXT("true") : TEXT("false"),
					MercenaryResult.Score,
					MercenaryResult.EarnedGold,
					MercenaryResult.Seed,
					MercenaryResult.PlaySeconds);
			}
		}
	}

	{
		CDR_Contracts::IScoreModeContract* const ScoreContractPtr = ProviderPtr->GetScoreModeContract();
		if (ScoreContractPtr != nullptr)
		{
			CDR_Contracts::FScoreRunResult ScoreResult;
			const CDR_Contracts::EConsumeResult ConsumeResult = ScoreContractPtr->ConsumeLastResult(ScoreResult);
			if (ConsumeResult == CDR_Contracts::EConsumeResult::Success)
			{
				UE_LOG(LogTemp, Log, TEXT("[MainHub] Consumed Score result: Cleared=%s Score=%d Gold=%d Seed=%d Seconds=%.2f"),
					(ScoreResult.bCleared == true) ? TEXT("true") : TEXT("false"),
					ScoreResult.Score,
					ScoreResult.EarnedGold,
					ScoreResult.Seed,
					ScoreResult.PlaySeconds);
			}
		}
	}
}

bool AMainHubGameMode::TryEnterMercenaryMode(UWorld* const WorldPtr, const FName SelectedModeIdValue)
{
	if (SelectedModeIdValue != NAME_Mode_Mercenary)
	{
		return false;
	}

	UGameInstance* const GameInstancePtr = WorldPtr->GetGameInstance();
	if (GameInstancePtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MainHub] Mercenary: GameInstance is null"));
		return true;
	}

	CDR_Contracts::IModeContractsProvider* const ProviderPtr = Cast<CDR_Contracts::IModeContractsProvider>(GameInstancePtr);
	if (ProviderPtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MainHub] Mercenary: GameInstance does not implement IModeContractsProvider"));
		return true;
	}

	CDR_Contracts::IMercenaryModeContract* const ContractPtr = ProviderPtr->GetMercenaryModeContract();
	if (ContractPtr == nullptr)
	{
		UE_LOG(LogTemp, Error, TEXT("[MainHub] Mercenary: Contract is null"));
		return true;
	}

	CDR_Contracts::FMercenaryStartContext StartContext;
	StartContext.Seed = 0;
	StartContext.Difficulty = 0;

	ContractPtr->SetStartContext(StartContext);

	UGameplayStatics::OpenLevel(WorldPtr, NAME_Level_MercenaryEntry);
	return true;
}

{
	if (SelectedModeIdValue != NAME_Mode_Score)

		return false;
	}

	UGameInstance* const GameInstancePtr = WorldPtr->GetGameInstance();

	{
		UE_LOG(LogTemp, Error, TEXT("[MainHub] Score: Contract is null"));
		return true;

	CDR_Contracts::FScoreStartContext StartContext;
	StartContext.Seed = 0;
	StartContext.Difficulty = 0;

	return true;
