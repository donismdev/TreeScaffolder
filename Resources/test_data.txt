@@@FILE_BEGIN {{Root}}/GameplayCore/Private/Random/CDRRandomManager.cpp
#include "Random/CDRRandomManager.h"

namespace CDR_Random
{
	static int32 ClampPercent(const int32 Value)
	{
		if (Value < 0)
		{
			return 0;
		}

		if (Value > 100)
		{
			return 100;
		}

		return Value;
	}

	void FRandomManager::Initialize(const int32 BaseSeedValue)
	{
		BaseSeed = BaseSeedValue;

		for (uint8 i = 0; i < static_cast<uint8>(ERandomCategory::Max); i += 1)
		{
			const ERandomCategory CategoryValue = static_cast<ERandomCategory>(i);
			const int32 DerivedSeedValue = MakeDerivedSeed(BaseSeedValue, CategoryValue);

			Streams[i].Initialize(DerivedSeedValue);
		}
	}

	int32 FRandomManager::GetBaseSeed() const
	{
		return BaseSeed;
	}

	int32 FRandomManager::RandRangeInt(const ERandomCategory CategoryValue, const int32 MinValue, const int32 MaxValue)
	{
		FRandomStream& StreamRef = GetStreamRef(CategoryValue);
		return StreamRef.RandRange(MinValue, MaxValue);
	}

	float FRandomManager::RandRangeFloat01(const ERandomCategory CategoryValue)
	{
		FRandomStream& StreamRef = GetStreamRef(CategoryValue);
		return StreamRef.FRand();
	}

	bool FRandomManager::RandBool(const ERandomCategory CategoryValue, const int32 TruePercentValue)
	{
		const int32 Clamped = ClampPercent(TruePercentValue);
		const int32 Roll = RandRangeInt(CategoryValue, 0, 99);

		if (Roll < Clamped)
		{
			return true;
		}

		return false;
	}

	int32 FRandomManager::MakeDerivedSeed(const int32 BaseSeedValue, const ERandomCategory CategoryValue)
	{
		// Deterministic, cheap, stable
		const int32 CategorySalt = 0x1F123BB5 + (static_cast<int32>(CategoryValue) * 0x9E3779B9);
		const int32 Mixed = (BaseSeedValue ^ CategorySalt) + 0x7F4A7C15;

		return Mixed;
	}

	FRandomStream& FRandomManager::GetStreamRef(const ERandomCategory CategoryValue)
	{
		const uint8 Idx = static_cast<uint8>(CategoryValue);

		if (Idx >= static_cast<uint8>(ERandomCategory::Max))
		{
			return Streams[static_cast<uint8>(ERandomCategory::Common)];
		}

		return Streams[Idx];
	}
}

{
	void FTurnBasedFlowStrategy::SetContext(FGameFlowContext* const InContextPtr)
	{
		ContextPtr = InContextPtr;
	}

	void FTurnBasedFlowStrategy::Initialize(UWorld* const WorldPtr)
	{
		(void)WorldPtr;

		Phase = EFlowPhase::Running;
		bWaitingForInput = true;
		CommandsAppliedThisTurn = 0;

		// NOTE:
		// Boards are NOT created here.
		// Each ModeFlow is responsible for creating boards with its own IDs and role.
	}

	void FTurnBasedFlowStrategy::Shutdown()
	{
		Phase = EFlowPhase::Finished;
		PendingCommands.Reset();
	}

	void FTurnBasedFlowStrategy::SubmitCommands(const FCommandBuffer& InCommands)
	{
		PendingCommands = InCommands;
	}

	void FTurnBasedFlowStrategy::Step(const int32 TickBudget)
	{
		(void)TickBudget;

		if (ContextPtr == nullptr)
		{
			Phase = EFlowPhase::Finished;
			return;
		}

		if (Phase != EFlowPhase::Running)
		{
			return;
		}

		if (PendingCommands.Commands.Num() <= 0)
		{
			bWaitingForInput = true;
			return;
		}

		bWaitingForInput = false;

		TArray<CDR_GameBoard::FBoardEvent> AllEvents;

		for (const CDR_GameBoard::FBoardCommand& Cmd : PendingCommands.Commands)
		{
			if (CommandsAppliedThisTurn >= RequiredCommandsPerTurn)
			{
				break;
			}

			CDR_GameBoard::IGameBoardState* const BoardPtr = ContextPtr->BoardSet.FindById(Cmd.TargetBoardId);
			if (BoardPtr == nullptr)
			{
				continue;
			}

			if (BoardPtr->CanAcceptCommand(Cmd) == false)
			{
				continue;
			}

			const bool bApplied = BoardPtr->ApplyCommand(Cmd);
			if (bApplied == true)
			{
				CommandsAppliedThisTurn += 1;
			}

			BoardPtr->CollectBoardEvents(AllEvents);
		}

		PendingCommands.Reset();

		if (ContextPtr->WorldState != nullptr)
		{
			CDR_World::FWorldReactionApplier::ApplyBoardEvents(AllEvents, *ContextPtr->WorldState);
		}

		if (CommandsAppliedThisTurn >= RequiredCommandsPerTurn)
		{
			CommandsAppliedThisTurn = 0;

			for (const TUniquePtr<CDR_GameBoard::IGameBoardState>& Board : ContextPtr->BoardSet.Boards)
			{
				if (Board.IsValid() == false)
				{
					continue;
				}

				Board->Simulate(0);
			}
		}
	}

	EFlowPhase FTurnBasedFlowStrategy::GetPhase() const
	{
		return Phase;
	}

	bool FTurnBasedFlowStrategy::IsWaitingForInput() const
	{
		return bWaitingForInput == true;
	}
}

#include "Flow/ModeScoreFlow.h"

	static CDR_GameBoard::FGameBoardId MakeScoreBoardId()
	{
		return CDR_GameBoard::FGameBoardId::Make(
			CDR_GameBoard::EGameBoardRole::Score,
			0,
			CDR_GameBoard::EGameBoardKind::CuboBoard,
			FName(TEXT("CuboBoard_0"))
		);

	void FModeScoreFlow::SetStartContext(const CDR_Contracts::FScoreStartContext& InStartContext)
	{
		StartContext = InStartContext;
	}

	void FModeScoreFlow::Initialize(UWorld* const WorldPtr)
	{
		CachedWorldPtr = WorldPtr;
		bWantsExitToMainHub = false;

		FlowRunner = MakeUnique<CDR_Flow::FGameFlowRunner>();
		if (FlowRunner == nullptr)
		{
			bWantsExitToMainHub = true;
			return;
		}

		TUniquePtr<CDR_Flow::IGameFlowStrategy> Strategy = MakeUnique<CDR_Flow::FTurnBasedFlowStrategy>();
		FlowRunner->Initialize(WorldPtr, MoveTemp(Strategy));

		CDR_Flow::FGameFlowContext* const CtxPtr = FlowRunner->GetMutableContext();
		if (CtxPtr == nullptr)
		{
			bWantsExitToMainHub = true;
			return;
		}

		// Seed must be applied ONCE here. All randomness must go through RandomManager.
		CtxPtr->InitializeRandom(StartContext.Seed);

		// Score mode board is created by mode (not by strategy).
		CtxPtr->BoardSet.Boards.Reset();
		CtxPtr->BoardSet.Boards.Add(MakeUnique<CDR_CuboBoard::FCuboBoardState>(MakeScoreBoardId()));

		Presentation = MakeUnique<FScorePresentationRoot>();
		if (Presentation != nullptr)
		{
			Presentation->Initialize(WorldPtr);
		}

		SpawnCoinEntityAtTurnEnd();
	}

	void FModeScoreFlow::Tick(const float DeltaSeconds)
	{
		if (FlowRunner == nullptr)
		{
			bWantsExitToMainHub = true;
			return;
		}

		if (Presentation != nullptr)
		{
			Presentation->Tick(DeltaSeconds);

			if (Presentation->WantsExit() == true)
			{
				bWantsExitToMainHub = true;
				return;
			}
		}

		PumpInputIntoFlow();

		FlowRunner->Tick(DeltaSeconds);

		const CDR_Flow::FGameFlowContext* const CtxPtr = FlowRunner->GetContext();
		if (CtxPtr == nullptr)
		{
			return;
		}

		TArray<CDR_GameBoard::FBoardEvent> Events;
		for (const TUniquePtr<CDR_GameBoard::IGameBoardState>& Board : CtxPtr->BoardSet.Boards)
		{
			if (Board.IsValid() == false)
			{
				continue;
			}

			Board->CollectBoardEvents(Events);
		}

		ProcessBoardEventsForScore(Events);
	}

	void FModeScoreFlow::Shutdown()
	{
		if (Presentation != nullptr)
		{
			Presentation->Shutdown();
			Presentation.Reset();
		}

		if (FlowRunner != nullptr)
		{
			FlowRunner->Shutdown();
			FlowRunner.Reset();
		}

		CachedWorldPtr = nullptr;
	}

	bool FModeScoreFlow::WantsExitToMainHub() const
	{
		return bWantsExitToMainHub == true;
	}

	void FModeScoreFlow::PumpInputIntoFlow()
	{
		if (FlowRunner == nullptr)
		{
			return;
		}

		if (Presentation == nullptr)
		{
			return;
		}

		if (Presentation->HasPendingPlaceCommand() == false)
		{
			return;
		}

		CDR_GameBoard::FBoardCommand Cmd;
		if (Presentation->ConsumePlaceCommand(Cmd) == false)
		{
			return;
		}

		CDR_Flow::FCommandBuffer Buffer;
		Buffer.Commands.Add(Cmd);

		FlowRunner->SubmitCommands(Buffer);

		CDR_Flow::FGameFlowContext* const CtxPtr = FlowRunner->GetMutableContext();
		if (CtxPtr == nullptr)
		{
			return;
		}

		CDR_CuboBoard::FCuboBoardState* CuboPtr = nullptr;

		for (const TUniquePtr<CDR_GameBoard::IGameBoardState>& Board : CtxPtr->BoardSet.Boards)
		{
			if (Board.IsValid() == false)
			{
				continue;
			}

			if (Board->GetKind() == CDR_GameBoard::EGameBoardKind::CuboBoard)
			{
				CuboPtr = static_cast<CDR_CuboBoard::FCuboBoardState*>(Board.Get());
				break;
			}
		}

		if (CuboPtr == nullptr)
		{
			return;
		}

		if (CuboPtr->GetPlacedThisTurn() >= 3)
		{
			CuboPtr->ResetTurnCounter();
			SpawnCoinEntityAtTurnEnd();
		}
	}

	void FModeScoreFlow::ProcessBoardEventsForScore(const TArray<CDR_GameBoard::FBoardEvent>& InEvents)
	{
		for (const CDR_GameBoard::FBoardEvent& Ev : InEvents)
		{
			if (Ev.EventType != CDR_GameBoard::EBoardEventType::EntityCollected)
			{
				continue;
			}

			int32 AddedScore = 0;

			if (Ev.Payload.EntityType == CDR_GameBoard::EBoardEntityType::CoinBronze)
			{
				AddedScore = 10;
			}
			else if (Ev.Payload.EntityType == CDR_GameBoard::EBoardEntityType::CoinSilver)
			{
				AddedScore = 50;
			}
			else if (Ev.Payload.EntityType == CDR_GameBoard::EBoardEntityType::CoinGold)
			{
				AddedScore = 200;
			}
			else
			{
				AddedScore = 0;
			}

			if (Presentation != nullptr)
			{
				Presentation->AddScore(AddedScore);
			}
		}
	}

	void FModeScoreFlow::SpawnCoinEntityAtTurnEnd()
	{
		if (FlowRunner == nullptr)
		{
			return;
		}

		CDR_Flow::FGameFlowContext* const CtxPtr = FlowRunner->GetMutableContext();
		if (CtxPtr == nullptr)
		{
			return;
		}

		CDR_CuboBoard::FCuboBoardState* CuboPtr = nullptr;

		for (const TUniquePtr<CDR_GameBoard::IGameBoardState>& Board : CtxPtr->BoardSet.Boards)
		{
			if (Board.IsValid() == false)
			{
				continue;
			}

			if (Board->GetKind() == CDR_GameBoard::EGameBoardKind::CuboBoard)
			{
				CuboPtr = static_cast<CDR_CuboBoard::FCuboBoardState*>(Board.Get());
				break;
			}
		}

		if (CuboPtr == nullptr)
		{
			return;
		}

		// ScoreMode uses Common category only.
		CDR_Random::FRandomManager& Rng = CtxPtr->Random;

		const int32 Roll = Rng.RandRangeInt(CDR_Random::ERandomCategory::Common, 0, 99);

		CDR_GameBoard::EBoardEntityType EntityType = CDR_GameBoard::EBoardEntityType::CoinBronze;

		if (Roll >= 90)
		{
			EntityType = CDR_GameBoard::EBoardEntityType::CoinGold;
		}
		else if (Roll >= 60)
		{
			EntityType = CDR_GameBoard::EBoardEntityType::CoinSilver;
		}
		else
		{
			EntityType = CDR_GameBoard::EBoardEntityType::CoinBronze;
		}

		for (int32 TryIndex = 0; TryIndex < 64; TryIndex += 1)
		{
			const int32 X = Rng.RandRangeInt(CDR_Random::ERandomCategory::Common, 0, CDR_CuboBoard::DefaultBoardWidth - 1);
			const int32 Y = Rng.RandRangeInt(CDR_Random::ERandomCategory::Common, 0, CDR_CuboBoard::DefaultBoardHeight - 1);

			const FIntPoint Grid(X, Y);

			const bool bSpawned = CuboPtr->TrySpawnEntityAtEmptyCell(Grid, EntityType);
			if (bSpawned == true)
			{
				CDR_GameBoard::FBoardEvent Ev;
				Ev.SourceBoardId = CuboPtr->GetBoardId();
				Ev.EventType = CDR_GameBoard::EBoardEventType::EntitySpawned;
				Ev.Payload.Grid = Grid;
				Ev.Payload.EntityType = EntityType;

				CuboPtr->EnqueueEvent(Ev);
				break;
			}
		}
	}
}

#include "Presentation/ScorePresentationRoot.h"

	void FScorePresentationRoot::Initialize(UWorld* WorldPtr)
	{
		CachedWorld = WorldPtr;
		Score = 0;
		bWantsExit = false;

		if (CachedWorld == nullptr)
		{
			return;
		}

		APlayerController* const PC = CachedWorld->GetFirstPlayerController();
		if (PC == nullptr)
		{
			return;
		}

		HUDWidget = CreateWidget<UScoreHUDWidget>(PC, UScoreHUDWidget::StaticClass());
		if (HUDWidget != nullptr)
		{
			HUDWidget->AddToViewport(0);
		}

		Presenter = MakeUnique<FScorePresenter>();
		if (Presenter != nullptr)
		{
			Presenter->Bind(HUDWidget, this);
		}

		PC->bShowMouseCursor = true;
		PC->SetInputMode(FInputModeGameAndUI());
	}

	void FScorePresentationRoot::Tick(const float DeltaSeconds)
	{
		(void)DeltaSeconds;

		if (Presenter != nullptr)
		{
			Presenter->Tick();

			if (Presenter->WantsExit() == true)
			{
				bWantsExit = true;
			}

			if (Presenter->HasPendingRandomPlace() == true)
			{
				// We only "request place".
				// Deterministic random grid is decided inside ModeScoreFlow (RandomManager).
				bHasPendingPlace = true;
			}
		}
	}

	void FScorePresentationRoot::Shutdown()
	{
		if (Presenter != nullptr)
		{
			Presenter->Unbind();
			Presenter.Reset();
		}

		if (HUDWidget != nullptr)
		{
			HUDWidget->RemoveFromParent();
			HUDWidget = nullptr;
		}

		CachedWorld = nullptr;

		bWantsExit = false;
		bHasPendingPlace = false;
	}

	bool FScorePresentationRoot::WantsExit() const
	{
		return bWantsExit == true;
	}

	bool FScorePresentationRoot::HasPendingPlaceCommand() const
	{
		return bHasPendingPlace == true;
	}

	bool FScorePresentationRoot::ConsumePlaceCommand(CDR_GameBoard::FBoardCommand& OutCommand)
	{
		if (bHasPendingPlace == false)
		{
			return false;
		}

		bHasPendingPlace = false;

		OutCommand = CDR_GameBoard::FBoardCommand{};
		OutCommand.CommandType = CDR_GameBoard::EBoardCommandType::Place;
		OutCommand.Payload.Grid = PendingPlaceGrid;
		OutCommand.Cost.TickCost = 1;

		OutCommand.TargetBoardId = CDR_GameBoard::FGameBoardId::Make(
			CDR_GameBoard::EGameBoardRole::Score,
			0,
			CDR_GameBoard::EGameBoardKind::CuboBoard,
			FName(TEXT("CuboBoard_0"))
		);

		return true;
	}

	void FScorePresentationRoot::AddScore(const int32 AddedScore)
	{
		Score += AddedScore;

		if (Presenter != nullptr)
		{
			Presenter->SetScore(Score);
		}
	}
