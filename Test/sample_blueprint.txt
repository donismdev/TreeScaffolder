#include "CoreMinimal.h"

#include "CDRSceneRouteTypes.generated.h"

USTRUCT()
struct FCDRSceneTransitionRequest final
{
	GENERATED_BODY()

public:
	UPROPERTY()
	ECDRSceneId From = ECDRSceneId::None;

	UPROPERTY()
	ECDRSceneId To = ECDRSceneId::None;

	UPROPERTY()
	FName Reason = NAME_None;

	bool IsValid() const
	{
		if (From == ECDRSceneId::None)
		{
			return false;
		}

		if (To == ECDRSceneId::None)
		{
			return false;
		}

		return true;
	}
};

@@@FILE_BEGIN {{Root}}/CDRGameDefs/Private/SceneRoute/CDRSceneRouterSubsystem.cpp
#include "SceneRoute/CDRSceneRouterSubsystem.h"

DEFINE_LOG_CATEGORY_STATIC(LogCDRSceneRouter, Log, All);

const TCHAR* UCDRSceneRouterSubsystem::ToSceneName(const ECDRSceneId InSceneId)
{
	switch (InSceneId)
	{
	case ECDRSceneId::MainHub: return TEXT("MainHub");
	case ECDRSceneId::ScoreMode: return TEXT("ScoreMode");
	case ECDRSceneId::MercenaryMode: return TEXT("MercenaryMode");
	default: return TEXT("None");
	}
}

bool UCDRSceneRouterSubsystem::CanTransition(const ECDRSceneId FromSceneId, const ECDRSceneId ToSceneId) const
{
	if (FromSceneId == ECDRSceneId::None)
	{
		return false;
	}

	if (ToSceneId == ECDRSceneId::None)
	{
		return false;
	}

	if (FromSceneId == ECDRSceneId::MainHub)
	{
		if (ToSceneId == ECDRSceneId::ScoreMode)
		{
			return true;
		}
		if (ToSceneId == ECDRSceneId::MercenaryMode)
		{
			return true;
		}
		return false;
	}

	if (FromSceneId == ECDRSceneId::ScoreMode)
	{
		if (ToSceneId == ECDRSceneId::MainHub)
		{
			return true;
		}
		return false;
	}

	if (FromSceneId == ECDRSceneId::MercenaryMode)
	{
		if (ToSceneId == ECDRSceneId::MainHub)
		{
			return true;
		}
		return false;
	}

	return false;
}

bool UCDRSceneRouterSubsystem::BuildTransitionRequest(
	const ECDRSceneId FromSceneId,
	const ECDRSceneId ToSceneId,
	const FName Reason,
	FCDRSceneTransitionRequest& OutRequest
) const
{
	OutRequest = FCDRSceneTransitionRequest();

	const bool bAllowed = CanTransition(FromSceneId, ToSceneId);
	if (bAllowed == false)
	{
		UE_LOG(LogCDRSceneRouter, Warning, TEXT("Blocked Transition: %s -> %s"),
			ToSceneName(FromSceneId),
			ToSceneName(ToSceneId)
		);
		return false;
	}

	OutRequest.From = FromSceneId;
	OutRequest.To = ToSceneId;
	OutRequest.Reason = Reason;

	UE_LOG(LogCDRSceneRouter, Log, TEXT("Transition OK: %s -> %s (Reason=%s)"),
		ToSceneName(OutRequest.From),
		ToSceneName(OutRequest.To),
		*Reason.ToString()
	);

	return (OutRequest.IsValid() == true);
}

@@@FILE_BEGIN {{Root}}/ScoreMode/Private/ScoreModeCore.cpp
#include "ScoreModeCore.h"
#include "Engine/World.h"
#include "SceneRoute/CDRSceneRouterSubsystem.h"

DEFINE_LOG_CATEGORY_STATIC(LogScoreModeCore, Log, All);

void UScoreModeCore::RequestExitToMainHub()
{
	if (CachedWorld == nullptr)
	{
		return;
	}

	UGameInstance* const GI = CachedWorld->GetGameInstance();
	if (GI == nullptr)
	{
		return;
	}

	UCDRSceneRouterSubsystem* const Router = GI->GetSubsystem<UCDRSceneRouterSubsystem>();
	if (Router == nullptr)
	{
		return;
	}

	FCDRSceneTransitionRequest Req;

	const bool bBuilt = Router->BuildTransitionRequest(
		ECDRSceneId::ScoreMode,
		ECDRSceneId::MainHub,
		FName(TEXT("ScoreRunCompleted")),
		Req
	);

	if (bBuilt == true)
	{
		UE_LOG(LogScoreModeCore, Log, TEXT("ScoreMode requested MainHub transition"));
	}
}
